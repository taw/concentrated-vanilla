		;
		;	scaling_data	[c]					- for fixed scaling type		- scale_factor = c
		;	scaling_data	[m	c]				- for linear scaling type		- scale factor = m*t + c
		;	scaling_data	[a	b	c]			- for quadratic scaling type	- scale factor = a*t^2 + b*t +c
		;	scaling_data	[a	b	c	d]		- for cubic scaling type		- scale factor = a*t^3 + b*t^2 + c*t + d
		;	scaling_data	[a	b	c	d]		- for bowl scaling type			- scale factor = (a*t^2 - b*t^3) * c + d
